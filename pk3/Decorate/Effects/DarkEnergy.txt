ACTOR DarkCrystalProjectile
{
    Radius 3
    Height 4
    Speed 45
    Damage 16
    SeeSound ""
    DeathSound ""
    RenderStyle Add
    Projectile
    Scale 0.5
    Alpha 3
    +DONTSPLASH
    +RANDOMIZE
    States
    {
        Spawn:
            DMBL ABCD 1          
            Loop
        Death:
            B128 A 1 A_PlaySound("Explosions/DarkEnergy", 5)
            TNT1 A 0 A_Explode(130,128,1)
            Stop
    }
}

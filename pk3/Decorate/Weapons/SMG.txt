ACTOR SMG : DoomWeapon
{
    Weapon.SlotNumber 4
    Weapon.AmmoUse 1
    Weapon.AmmoGive 150
    Weapon.AmmoType "Clip5mm"
    Inventory.PickupMessage "Got an SMG chambered in 5mm bullets"
    States
    {
        Ready:
            SMGF A 1 A_WeaponReady
            Loop
        Deselect:
            SMGF A 0 A_PlaySound("12mmPistol/Lower", CHAN_WEAPON)
            Goto Lower
        Lower:
            SMGF A 1 A_Lower
            Loop
        Select:
            SMGF A 0 A_PlaySound("12mmPistol/Raise", CHAN_WEAPON)
            Goto Raise
        Raise:
            SMGF A 1 A_Raise
            Loop
        Fire:
            SMGF A 0 A_PlaySound ("SMG/Fire", CHAN_WEAPON)
            SMGF B 1 A_FireBullets(2.4, 0, 1, 3, "BulletPuff")
            SMGF C 1
            SMGF E 1
            SMGF F 1 A_ReFire
            Goto Ready
        Spawn:
        SMGI A -1
        Stop
    }
}

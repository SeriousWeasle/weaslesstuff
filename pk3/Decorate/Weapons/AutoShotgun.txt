ACTOR AutoShotgun : DoomWeapon
{
    Weapon.SlotNumber 3
    Weapon.AmmoUse 1
    Weapon.AmmoGive 12
    Weapon.AmmoType "ShellAuto"
    Inventory.PickupMessage "Got the Auto Shotgun"
    States
    {
        Ready:
            ASHT A 1 A_WeaponReady
            Loop
        Deselect:
            ASHT A 1 A_Lower
            Loop
        Select:
            ASHT A 1 A_Raise
            Loop
        Fire:
            ASHT A 2 A_PlaySound("AutoShotgun/Fire", CHAN_WEAPON)
            ASHT B 1
            ASHT C 6 A_FireBullets(5.6, 0, 7, 2, "BulletPuff")
            ASHT A 2 A_ReFire
            ASHT A 0 A_PlaySound("AutoShotgun/Pump", 5)
            ASHT DEFED 3
            Goto Ready
        Spawn:
            SHOT A -1
            Stop
    }
}

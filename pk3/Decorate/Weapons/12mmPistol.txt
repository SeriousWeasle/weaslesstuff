ACTOR 12mmPistol : DoomWeapon
{
    Weapon.SlotNumber 2
    Weapon.AmmoUse1 1
    Weapon.AmmoGive1 24
    Weapon.AmmoType1 "Clip12mm"
    Weapon.AmmoUse2 1
    Weapon.AmmoGive2 2
    Weapon.AmmoType2 "Clip12mm"
    Inventory.PickupMessage "Got a CE standard issue 12 mm Pistol"
    States
    {
        Ready:
            PSTV A 1 A_WeaponReady
            Loop
        Deselect:
            PSTV A 0 A_PlaySound("12mmPistol/Lower", CHAN_WEAPON)
            Goto Lower
        Lower:
            PSTV A 1 A_Lower
            Loop
        Select:
            PSTV A 0 A_PlaySound("12mmPistol/Raise", CHAN_WEAPON)
            Goto Raise
        Raise:
            PSTV A 1 A_Raise
            Loop
        Fire:
            PSTV A 2 A_PlaySound ("12mmPistol/Fire", CHAN_WEAPON)
            PSTV B 7 A_FireBullets(2.4, 0, 1, 16, "BulletPuff")
            PSTV C 5
            PSTV D 5
            PSTV E 5
            PSTV F 6 A_ReFire
            Goto Ready
        AltFire:
            PSTV A 1 A_PlaySound ("12mmPistol/Fire")
            PSTV B 4 A_FireBullets(5.8, 0, 1, 16, "BulletPuff")
            PSTV C 2 
            PSTV D 2
            PSTV E 2
            PSTV F 3 A_ReFire
            Goto Ready
        Spawn:
        PSTI A -1
        Stop
    }
}

            PLCN C 1
            PLCN D 1
            PLCN AA 1
            PLCN BB 1
            PLCN CC 1
            PLCN DDD 1
            PLCN AAA 1
            PLCN BBB 1
            PLCN CCC 1
            PLCN DDDD 1 A_TakeInventory("PlasmaCannonSpinUp", 1)
